Problem:

Create a function that determines whether or not a player is holding a Full House in their hand. A hand is represented as an array of 5 cards. A full house is defined as a pair of cards and a three-of-a-kind.
To illustrate: [“A”, “A”, “A”, “K”, “K”] would be a Full House, since the player holds 3 aces and 2 kings.
Examples
isFullHouse([“A”, “A”, “A”, “K”, “K”]) ➞ true
isFullHouse([“3”, “J”, “J”, “3”, “3”]) ➞ true
isFullHouse([“10”, “J”, “10”, “10”, “10”]) ➞ false
isFullHouse([“7”, “J”, “3”, “4”, “2”]) ➞ false
Notes
N/A

My solution:

#include

using namespace std;

bool isFullHouse ( char arr[], int size );

int main() {

char deck[5] = {'K', 'K', 'K', 'L', 'Q'};

bool check = isFullHouse(deck, 5);
if (check) {
    cout << "deck is a full house" << endl;
} else {
    cout << "deck is not a full house" << endl;
}

return 0;

}

bool isFullHouse ( char arr[], int size ) {

char unique_[size];

int count = 0;

for ( int i = 0; i < size ; i++ ){
    bool tmp = false;
    for ( int x = 0; x < i; x++ ){
        if ( arr[x] == arr[i] ) {
            tmp = true;
            break;
        }
    }

    if (!tmp){
        count++;
    }
}

if ( count > 2 ) {
    return false;
}

char card1 = arr[0], card2;
int i;

for ( i = 0 ; i < size ; i++ ) {
    if ( arr[i] != card1 ) {
        card2 = arr[i];
    }
}

if ( i == size - 1) {
    return false;
}

int card1_count = 0, card2_count = 0;

for ( i = 0 ; i < size ; i++ ) {
    if (arr[i] == card1) {
        card1_count++;
    }
}

if ( card1_count != 2 && card1_count != 3 ) {
    return false;
}

for ( i = 0 ; i < size ; i++ ) {
    if ( arr[i] == card2 ) {
        card2_count++;
    }
}

if ( card2_count == size - card1_count ) {
    return true;
} else {
    return false;
}

}
